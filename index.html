🔌 Настройка Supabase

Выполните этот SQL в редакторе Supabase (SQL Editor → New query), чтобы создать нужные таблицы.

-- 1. Создайте проект на supabase.com -- 2. Откройте SQL Editor и вставьте этот код: create table if not exists or_shtat ( id text primary key, data jsonb not null default '{}', updated_at timestamptz default now() ); create table if not exists or_gph ( id text primary key, data jsonb not null default '{}', updated_at timestamptz default now() ); create table if not exists or_ref ( key text primary key, values jsonb not null default '[]' ); -- Включить Row Level Security alter table or_shtat enable row level security; alter table or_gph enable row level security; alter table or_ref enable row level security; -- Разрешить чтение/запись через anon-ключ create policy "anon_all" on or_shtat for all using (true) with check (true); create policy "anon_all" on or_gph for all using (true) with check (true); create policy "anon_all" on or_ref for all using (true) with check (true); -- Контент-план create table if not exists or_content ( id text primary key, data jsonb not null default '{}', updated_at timestamptz default now() ); alter table or_content enable row level security; create policy "anon_all" on or_content for all using (true) with check (true); -- Таблица бекапов create table if not exists or_backups ( id text primary key, created_at timestamptz default now(), label text, shtat_count integer default 0, gph_count integer default 0, data jsonb not null default '{}' ); alter table or_backups enable row level security; create policy "anon_all" on or_backups for all using (true) with check (true); -- ⚡ Realtime-синхронизация между пользователями -- Выполните отдельно если таблицы уже существуют: alter publication supabase_realtime add table or_shtat; alter publication supabase_realtime add table or_gph; alter publication supabase_realtime add table or_content; alter publication supabase_realtime add table or_ref;

После выполнения SQL нажмите «🔌 База данных» в боковом меню и введите URL и anon-ключ проекта.
